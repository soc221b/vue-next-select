Getting started

Installation

via npm

npm add vue-next-select

via CDN

<script src="https://unpkg.com/vue-next-select"></script>
<link rel="stylesheet" href="https://unpkg.com/vue-next-select/dist/index.min.css">

Usage

import { createApp, ref } from 'vue'
import VueNextSelect from 'vue-next-select'
import 'vue-next-select/dist/index.min.css'

const app = createApp({
  name: 'app',
  setup () {
    const model = ref(null)
    const options = ['vue', 'next', 'select']
    return {
      model,
      options
    }
  },
  template: `
    <vue-next-select
      v-model="model"
      :options="options"
    ></vue-next-select>
  `
})

app.component('vue-select', VueNextSelect)
app.mount('#app')
